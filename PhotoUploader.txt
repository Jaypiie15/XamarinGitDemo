    private async Task  UploadPhoto(int productId)
        {
            var client = new RestClient(apiUrl);
            var request = new RestRequest("product/displayphoto/"+productId.ToString(), DataFormat.Json);

            request.AlwaysMultipartFormData = true;
            request.Method = Method.POST;

            request.AddFileBytes("test.jpg", byteArray, "test.jpg", "image/jpg");
            var response = await client.ExecuteAsync(request);
        };
